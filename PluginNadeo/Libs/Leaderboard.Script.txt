#Include "Libs/Nadeo/Manialink2.Script.txt"						as Manialink
#Include "Libs/Nadeo/TMConsole/Menu/MenuLayers.Script.txt" 		as Layers

#Const C_LayerName	"Layer_TMC_Plugin_Leaderboard"



Void CreateLayer()
{
	declare ML = 
	"""
	<frame>
	</frame>
	
	
	
	<script><!--
		{{{Manialink::GetIncludes()}}}
		#Const DebugName "L_PluginLeaderboard"
		{{{Manialink::Load()}}}
		main() 
		{
			log(Now^" > LOAD OK : {{{C_LayerName}}}");
			while(True)
			{
				yield;
				if(!PageIsVisible) continue;
				LibManialink_AnimLoop();
			}
		}
	--></script>
	""";
	
	
	Layers::Create(C_LayerName, ML);
	Layers::Attach(C_LayerName);
}

Void Load()
{
	CreateLayer();
}

CUILayer GetLayer()
{
	return Layers::Get(C_LayerName);
}

Text GetLayerName()
{
	return C_LayerName;
}
